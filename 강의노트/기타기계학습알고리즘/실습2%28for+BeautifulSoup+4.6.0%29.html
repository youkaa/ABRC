	function GetCommentList(id, page, order) {

	    $.ajax({
	        type: "GET",
	        url: CROSSDOMAIN_URL,
	        contentType: "application/json",
	        data: {
	            Url: "http://webapi2.edaily.co.kr/Edaily/RC/Front/COMMENTLIST",
	            Page: page,
	            Page_Size: "3",
	            News_ID: id,
	            News_GB: "E",
	            orderBy: order,
	            R_WhereopenID: "",
	            R_LoginID: ""
	        },
	        success: function (result) {
	            result = JSON.parse(result);
	            //뉴스페이지냐 코멘트보기페이지냐 정렬규칙이 바꼈느냐에 따라 영역클리어 구분
	            if ("0" > 0) {
	                //정렬이 바꼈으면 영역클리어 후 정렬값세팅
	                if ($("#OrderBy").val() != order) {
	                    $("#CommentBindArea").html("");
	                    $("#OrderBy").val(order);
	                }
	            }
	            else {
	                //뉴스읽기페이지에선 정렬조건 변경만 존재하므로 무조건 영역클리어
	                $("#CommentBindArea").html("");
	            }

	            var arrComment = [];
	            var arrReply = [];

	            if (result.C_List.length > 0) {
	                $(".not_reply").hide();

	                for (var i = 0; i < result.C_List.length; i++) {
	                    if (result.C_List[i].C_Seq == result.C_List[i].C_Ref) {
	                        arrComment.push(result.C_List[i]);
	                    }
	                    else {
	                        arrReply.push(result.C_List[i]);
	                    }
	                }
                    
	                console.log(result);
	                console.log(arrComment);
	                $("#CommentsBind").tmpl(arrComment).appendTo($("#CommentBindArea"));

	                if (arrReply.length > 0) {
	                    for (var j = 0; j < arrComment.length; j++) {
	                        var arrSub = [];
	                        for (i = 0; i < arrReply.length; i++) {
	                            if (arrComment[j].C_Seq == arrReply[i].C_Ref) {
	                                arrSub.push(arrReply[i]);
	                            }
	                        }
	                        $("#CommentSub").tmpl(arrSub).appendTo($("#Reply" + arrComment[j].C_Ref));
	                    }
	                }

	                $("#TotalCommentCNT").val(result.TotalCount.toString());

					// 독자의견 로그아웃 추가 || 2016.11.02 이찬희
					var snsId = getCookie("snsId");
					if (snsId == "" || snsId == null || snsId == undefined)
					{

						$(".tlt_opinion").html("<span class='sns_txt_count'>독자의견</span><em>" + result.TotalCount + "</em><p class='right'><button type='button' class='btn_log' onclick='deleteCookieSNS()'>로그아웃</button></p>");
					}

	                $("#emCommentCount").text(result.TotalCount);

	                //더보기 버튼 설정
	                if (result.TotalCount > (parseInt($("#Page").val()) * parseInt($("#PageSize").val()))) {
	                    $(".btn_opinionAll").show();

	                    if ("0" > 0) {
	                        $(".btn_opinionAll").html("더보기<span class='blt_arrow'></span>");
	                    }
	                    else {
	                        $(".commentcnt").html(result.TotalCount);
	                    }

	                }
	                else {
	                    $(".btn_opinionAll").hide();
	                }
	            }
	            else {
	                $(".btn_opinionAll").hide();
	                $(".not_reply").html("등록된 의견이 없습니다.");
	                $(".not_reply").show();
	            }

	            //답글
	            if ($('.reply_lowArea').is(':visible')) {
	                $('.reply_lowArea').hide();
	            }

	            //정렬순 index클래스처리
	            $("#orderbyLatest").removeClass('active');
	            $("#orderbyReply").removeClass('active');
	            $("#orderbyRecommand").removeClass('active');

	            console.log(error);

	function TextLimit (input, count, maximumByte) {
		var before = count.val() * 1;
		var str_len = input.val().length;
		var cbyte = 0;
		var li_len = 0;
		for (i = 0; i < str_len; i++) {
			var ls_one_char = input.val().charAt(i);
			if (escape(ls_one_char).length > 4) {
				cbyte += 2; //한글이면 2를 더한다
			} else {
				cbyte++; //한글아니면 1을 더한다
			}
			if (cbyte <= maximumByte) {
				li_len = i + 1;
			}
		}
		// 사용자가 입력한 값이 제한 값을 초과하는지를 검사한다.
		if (parseInt(cbyte) > parseInt(maximumByte)) {

		    alert("허용된 글자수 " + maximumByte + "byte 가 초과되었습니다.<br/>초과된 부분은 자동으로 삭제됩니다.");
		    var str = input.val();
		    var str2 = input.val().substr(0, li_len);
		    input.val(str2);
		    var cbyte = 0;
		    for (i = 0; i < input.val().length; i++) {
		        var ls_one_char = input.val().charAt(i);
		        if (escape(ls_one_char).length > 4) {
		            cbyte += 2; //한글이면 2를 더한다
		        } else {
		            cbyte++; //한글아니면 1을 다한다
		        }
		    }
		}
		count.text(cbyte);
	}

    news = soup.find(id="viewcontent_inner")
    # <br> 태그 제거하기
    for br in news.find_all('br'):

    # <div> 태그 제거하기

    text = news.get_text()
